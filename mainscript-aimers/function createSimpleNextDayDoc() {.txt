function createSimpleNextDayDoc() {
  const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("Form Responses 1");
  const lastRow = sheet.getLastRow();
  const timezone = Session.getScriptTimeZone();

  const today = sheet.getRange(lastRow, 1).getValue(); // Column A (Timestamp)
  const planForTomorrow = sheet.getRange(lastRow, 23).getValue(); // Column W

  const nextDay = new Date(today);
  nextDay.setDate(nextDay.getDate() + 1);
  const formattedNextDate = Utilities.formatDate(nextDay, timezone, "yyyy-MM-dd");

  const docName = `Plan for ${formattedNextDate}`;
  const folder = DriveApp.getFolderById("114LTprcJz-R6CQ06K_ZMJ6qroDMOD2Wb"); // Your folder ID
  const doc = DocumentApp.create(docName);
  folder.addFile(DriveApp.getFileById(doc.getId()));
  DriveApp.getRootFolder().removeFile(DriveApp.getFileById(doc.getId())); // Remove from root

  const body = doc.getBody();

  // Add simple plan text
  body.appendParagraph(`üóìÔ∏è Plan for ${formattedNextDate}`)
      .setHeading(DocumentApp.ParagraphHeading.HEADING1)
      .setAlignment(DocumentApp.HorizontalAlignment.CENTER);

  body.appendParagraph("\n" + planForTomorrow)
      .setFontSize(12)
      .setForegroundColor("#202124");

  doc.saveAndClose();
}
